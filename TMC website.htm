    let user = null;

    let cart = [];

    let total = 0;

 

    function showView(id) {

        document.querySelectorAll('.view').forEach(v => v.classList.remove('active'));

        document.getElementById(id).classList.add('active');

        if(id === 'home-view') renderShops();

        if(id === 'orders-view') renderOrders();

        window.scrollTo(0,0);

    }

 

    function login() {

        const email = document.getElementById('email').value;

        if(!email) return alert("Please enter your email");

        user = email;

        document.getElementById('nav-user').innerText = email;

        if(!localStorage.getItem(user)) localStorage.setItem(user, JSON.stringify({ orders: [] }));

        showView('home-view');

    }

 

    function renderShops() {

        const list = document.getElementById('shop-list');

        list.innerHTML = shops.map(s => `

            <div class="card" onclick="openMenu(${s.id})">

                </div>

            </div>

        `).join('');

    }

 

    function openMenu(id) {

        const shop = shops.find(s => s.id === id);

        document.getElementById('current-shop-name').innerText = shop.name;

        const list = document.getElementById('menu-list');

        list.innerHTML = shop.items.map(i => `

            <div class="menu-item">

                <img src="${i.img}" alt="${i.n}">

                <div class="menu-info">

                    <div style="font-weight: 700;">${i.n}</div>

                    <div style="color:var(--tmc-gray); font-size: 14px;">$${i.p.toFixed(2)}</div>

                </div>

                <button class="add-btn" onclick="addToCart('${i.n}', ${i.p})">+</button>

            </div>

        `).join('');

        showView('menu-view');

    }

 

    function addToCart(n, p) {

        cart.push(n);

        total += p;

        const priceStr = `$${total.toFixed(2)}`;

        document.getElementById('cart-cost').innerText = priceStr;

        document.getElementById('final-price').innerText = priceStr;

        document.getElementById('cart-float').style.display = 'block';

    }

 

    function placeOrder() {

        const addr = document.getElementById('addr').value;

        if(!addr || cart.length === 0) return alert("Please add an address and items to your cart");

       

        let userData = JSON.parse(localStorage.getItem(user));

        const order = {

            id: Math.floor(Math.random() * 90000) + 10000,

            items: cart.join(", "),

            total: total.toFixed(2),

            status: "Preparing",

            time: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })

        };

 

        userData.orders.push(order);

        localStorage.setItem(user, JSON.stringify(userData));

 

        setTimeout(() => {

            let data = JSON.parse(localStorage.getItem(user));

            let o = data.orders.find(x => x.id === order.id);

            if(o) o.status = "Delivered";

            localStorage.setItem(user, JSON.stringify(data));

        }, 8000);

 

        alert("Order #" + order.id + " has been placed!");

        cart = []; total = 0;

        document.getElementById('cart-float').style.display = 'none';

        showView('orders-view');

    }

 

    function renderOrders() {

        const list = document.getElementById('history-list');

        const userData = JSON.parse(localStorage.getItem(user));

       

        if(!userData.orders.length) {

            list.innerHTML = "<p style='text-align:center; color:#999;'>No active or past orders.</p>";

            return;

        }

 

        list.innerHTML = userData.orders.reverse().map(o => `

            <div style="background:white; padding:20px; margin-bottom:15px; border-radius:15px; border:1px solid #eee; position: relative;">

                <div style="display:flex; justify-content:space-between; margin-bottom: 10px;">

                    <span style="font-size: 12px; color: #999;">${o.time}</span>

                </div>

                <div style="font-weight: 700; margin-bottom: 5px;">Order #${o.id}</div>
